def setup
  createCanvas(500, 500)
  background("#D7D2D0")
end

def draw
  translate(width / 2, height / 2)
  rotate(PI / 2)

  stroke(0)
  fill("#000505")
  keigionrails2024symbol
end

def keigionrails2024symbol
  # Kaigi on Rails2024 Ruby Shape
  beginShape()
  vertex(-55, 35)
  vertex(-55, -35)
  vertex(0, -65)
  vertex(63, 0)
  vertex(0, 65)
  endShape(CLOSE)

  # spotlight for Rubyist
  beginShape()
  vertex(-55, -10)
  vertex(38, 25)
  vertex(0, 65)
  fill("#FF4053")
  endShape(CLOSE)

  # Ariakeblue spotlight
  beginShape()
  vertex(-55, -10)
  vertex(0, -65)
  vertex(33, -30)
  fill("#00C5DA")
  endShape(CLOSE)
end
